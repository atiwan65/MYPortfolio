# ต้องมี Database

ต้องมี mogoDB

สร้าง mogoDB แล้วเข้าไปสร้า Database แบบ Cloud

ไปที่ backend

ที่ไฟล์ db_config

ใส่ลื้งลงไปตรง url

module.exports = {
    url:"Cloud ของคุณ"
}
